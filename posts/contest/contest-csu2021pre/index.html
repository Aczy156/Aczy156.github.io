A

题目大意：贪心，可以无限选，然后问给定容积，最大价值。

思路永远选单位价值最大的，然后逐级取余。

#include <iostream>
using namespace std;
int a, ans;

int main() {
    scanf("%d", &a);
    ans += (a/16)*227;
    a %= 16;
    ans += (a/8)*109;
    a %= 8;
    ans += (a/4)*49;
    a %= 4;
    ans += (a/2)*23;
    a %= 2;
    ans += a*10;
    printf("%d\n", ans);
    return 0;
}
